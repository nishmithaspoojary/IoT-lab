const int dry=800;
const int wet=300;
int sensor=0;

void setup()
{
  pinMode(A0, INPUT);
  pinMode(13, OUTPUT);
  Serial.begin(9600);
}

void loop()
{
  int sensor=analogRead(A0);
  Serial.println(sensor);
  if(sensor>=dry){
    Serial.println("very dry");
    digitalWrite(13,HIGH);
}
  else if(sensor<dry && sensor>wet){
    Serial.println("Moderately dry");  }
  else{
    Serial.println("wet");
    digitalWrite(13,LOW);
  }
  delay(1000);
}
